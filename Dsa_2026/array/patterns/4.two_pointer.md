# Code (VERY IMPORTANT)

```javascript id="jlwm33"
function pairSum(arr, target) {

    let left = 0;
    let right = arr.length - 1;

    while (left < right) {

        let sum = arr[left] + arr[right];

        if (sum === target) {
            return [arr[left], arr[right]];
        }

        else if (sum < target) {
            left++;
        }

        else {
            right--;
        }
    }

    return null;
}

console.log(pairSum([1,2,3,4,6], 6));
```

Output:

```text id="jlwm34"
[2,4]
```

# Code

```javascript id="jlwm42"
function isPalindrome(str) {

    let left = 0;
    let right = str.length - 1;

    while (left < right) {

        if (str[left] !== str[right]) {
            return false;
        }

        left++;
        right--;
    }

    return true;
}

console.log(isPalindrome("madam"));
```

Output:

```text id="jlwm43"

```


```text id="jlwm47"

```

---

# Visualization

```text id="jlwm48"

```

# Code

```javascript id="jlwm49"
function removeDuplicates(arr) {

    let slow = 0;

    for (let fast = 1; fast < arr.length; fast++) {

        if (arr[fast] !== arr[slow]) {

            slow++;
            arr[slow] = arr[fast];
        }
    }

    return slow + 1;
}

const arr = [1,1,2,2,3];

const length = removeDuplicates(arr);

console.log(arr.slice(0, length));
```

Output:

```text id="jlwm50"
[1,2,3]
```

# Common Mistakes

---

## Mistake 1

Using two pointer on unsorted array.

Many pair problems REQUIRE sorting first.

---

## Mistake 2

Infinite loop.

Forgetting:

```text id="jlwm52"
left++
or
right--
```

---

## Mistake 3

Wrong loop condition.

Correct:

```javascript id="jlwm53"
while (left < right)
```
